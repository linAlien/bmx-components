# 基本使用


<bmx-inspect-task-form
      ref="taskForm"
></bmx-inspect-task-form>

onLoad(option) {
	  //{id, disabled}
    this.$nextTick(() => {
      this.$refs.taskForm.init(option);
    });
 },
